ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
    ListNode *node = nullptr;
    ListNode *iter = nullptr;
    
    int sum = 0;
    int nextAdd = 0;
    while (l1 && l2) {
        sum = l1->val + l2->val + nextAdd;
        nextAdd = sum / 10;
        sum = sum % 10;
        if(node == nullptr)
        {
            node = new ListNode(sum);
            iter = node;
        }
        else
        {
            iter->next = new ListNode(sum);
            iter = iter->next;
        }
        
        l1 = l1->next;
        l2 = l2->next;
    }
    
    ListNode *tmp = nullptr;
    if(l1 || l2)
    {
        tmp = l1 ? l1 : l2;
    }
    while(tmp)
    {
        sum = tmp->val + nextAdd;
        nextAdd = sum / 10;
        sum = sum % 10;
        
        if(node == nullptr)
        {
            node = new ListNode(sum);
            iter = node;
        }
        else
        {
            iter->next = new ListNode(sum);
            iter = iter->next;
        }
        
        tmp = tmp->next;
    }
    
    if(nextAdd)
    {
        iter->next = new ListNode(nextAdd);
    }
    
    return node;
}
